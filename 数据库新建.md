```mysql
DROP TABLE IF EXISTS `answer`;
DROP TABLE IF EXISTS `answersheet`;
DROP TABLE IF EXISTS `option`;
DROP TABLE IF EXISTS `question`;
DROP TABLE IF EXISTS `questionnaire`;
DROP TABLE IF EXISTS `user`;

CREATE TABLE `answer` (
  `paperid` int(11) NOT NULL,
  `questionid` int(11) NOT NULL,
  `ip` varchar(255) NOT NULL,
  `selectid` int(11) NOT NULL,
  `answer` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`paperid`,`questionid`,`ip`,`selectid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;

CREATE TABLE `answersheet` (
  `paperid` int(11) NOT NULL,
  `ip` varchar(255) NOT NULL,
  `anstime` datetime NOT NULL,
  PRIMARY KEY (`paperid`,`ip`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;

CREATE TABLE `option` (
  `selectid` int(11) NOT NULL,
  `questionid` int(11) NOT NULL,
  `goquestion` int(11) DEFAULT NULL,
  `scontent` varchar(255) NOT NULL,
  `paperid` int(11) NOT NULL,
  PRIMARY KEY (`selectid`,`questionid`,`paperid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;

CREATE TABLE `question` (
  `paperid` int(11) NOT NULL,
  `questionid` int(11) NOT NULL,
  `questiontitle` varchar(255) NOT NULL,
  `type` int(11) NOT NULL,
  `ismust` bit(1) NOT NULL,
  `min` int(11) DEFAULT NULL,
  `max` int(11) DEFAULT NULL,
  PRIMARY KEY (`paperid`,`questionid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;

CREATE TABLE `questionnaire` (
  `paperid` int(11) NOT NULL AUTO_INCREMENT,
  `papertitle` varchar(255) NOT NULL,
  `ispublish` bit(1) NOT NULL,
  `userid` int(11) NOT NULL,
  `createtime` datetime NOT NULL,
  PRIMARY KEY (`paperid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;

CREATE TABLE `user` (
  `userid` int(11) NOT NULL AUTO_INCREMENT,
  `password` varchar(255) NOT NULL,
  `username` varchar(255) NOT NULL,
  PRIMARY KEY (`userid`),
  UNIQUE(`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;
```
